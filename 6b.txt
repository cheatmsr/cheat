#include <stdio.h>

void ToH(int n, char source, char spare, char dest);
static int step = 0;

int main() {
    int n;
    printf("\nEnter the number of rings: ");
    scanf("%d", &n);
    ToH(n, 'A', 'B', 'C');
    return 0;
}

void ToH(int n, char source, char spare, char dest) {
    if (n == 1) {
        printf("\nStep %d: Move ring %d from %c to %c", ++step, n, source, dest);
    } else {
        ToH(n - 1, source, dest, spare);
        printf("\nStep %d: Move ring %d from %c to %c", ++step, n, source, dest);
        ToH(n - 1, spare, source, dest);
    }
}
